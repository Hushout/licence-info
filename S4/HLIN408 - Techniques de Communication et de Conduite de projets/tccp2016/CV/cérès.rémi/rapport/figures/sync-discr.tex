\tikzset{
    decision/.style={
        diamond, aspect=2, draw, fill=yellow!30,
        align=center, text badly centered, inner sep=0pt
    },
    block/.style={
        rectangle, draw, fill=blue!30,
        align=center, text centered, inner sep=.5em
    },
    startend/.style={
        block, rounded corners, fill=white,
        minimum width=3em, minimum height=2em
    },
    line/.style={
        draw, ->, >=stealth
    }
}

\begin{document}
\begin{tikzpicture}[node distance=.5cm]
    \node [startend]
        (start) {début};

    \node [block, below=of start]
        (version-dist) {Télécharger le numéro\\de version distant};

    \node [decision, below=of version-dist]
        (chgt-decision) {Changements\\locaux~?};

    \node [decision, below left=.5cm and 2.6cm of chgt-decision]
        (version-decision-1) {Numéro local $\gtreqless$\\numéro distant ?};

    \node [decision, below right=.5cm and 2.6cm of chgt-decision]
        (version-decision-2) {Numéro local $\gtreqless$\\numéro distant ?};

    \node [block, below=.5cm and 1cm of version-decision-1]
        (res-modif-dist) {Modification distante sans\\modification locale};

    \node [block] at (res-modif-dist-|version-decision-2)
        (res-conflit) {Conflit de modifications};

    \node [block] at ($(res-modif-dist)!0.5!(res-conflit)$)
        (res-modif-local) {Modification locale sans\\modification distante};

    \node [startend, yshift=-1.25cm]
        (fin) at (res-modif-dist-|chgt-decision) {fin};

    \path [line] (start) -- (version-dist);
    \path [line] (version-dist) -- (chgt-decision);
    \path [line] (chgt-decision) -| node [near start, above] {non}
        (version-decision-1);
    \path [line] (chgt-decision) -| node [near start, above] {oui}
        (version-decision-2);
    \path [line] (version-decision-2) -- node [midway, right] {$<$}
        (res-conflit);
    \path [line] (version-decision-2) -| node [near start, above] {$\geqslant$}
        (res-modif-local);
    \path [line] (version-decision-1) -- node [midway, right] {$<$}
        (res-modif-dist);
    \path [line] (version-decision-1.west) --++ (-.5, 0) |-
        node [pos=0.595, above] {$\geqslant$}
        (fin.west);
    \path [line] (res-modif-dist) |- (fin);
    \path [line] (res-modif-local) -- (fin);
    \path [line] (res-conflit) |- (fin);
\end{tikzpicture}
