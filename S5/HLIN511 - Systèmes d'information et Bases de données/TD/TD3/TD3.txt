SELECT NUM_AB, NOM, (TARIF * (REDUC/100))
FROM ABONNE
WHERE REDUC IS NOT NULL
AND (TARIF * (REDUC/100)) < 20;

//Q11
SELECT LIVRE.ISBN, TITRE
FROM LIVRE, EXEMPLAIRE
WHERE LIVRE.ISBN = EXEMPLAIRE.ISBN
AND CATEGORIE = 'ROMAN';

//Q12
SELECT ISBN, TITRE, SUM(NUMERO)
FROM LIVRE, EXEMPLAIRE
WHERE LIVRE.ISBN = EXEMPLAIRE.ISBN
AND CATEGORIE = 'EXCLU'
GROUP BY TITRE
